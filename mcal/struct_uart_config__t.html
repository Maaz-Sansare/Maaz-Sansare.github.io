Detailed Description

Uart configuration structure.

Example to configure UART:

UartHWConfig_t hw_config = {0};
hw_config.tx.port = PCF_PORT_0;
hw_config.tx.pin = PCF_PIN_1;
hw_config.rx.port = PCF_PORT_0;
hw_config.rx.pin = PCF_PIN_0;
UartConfig_t config = {0};
config.id = PCF_UART4_ID;
config.mode = enUartMode_Both; // TX + RX mode.
config.baudrate = 115200; // baudrate value set to 115200
config.databits = enUartDatabits_8; // data length is of 8 bits
config.parity = enUartParity_None; // parity bit is none
config.stopbits = enUartStopbits_One; // stop bit
config.fc = enUartFlowControl_None; // flow control to none
config.txBufferSize = 16; // Tx buffer size 16 (Non-Blocking mode)
config.rxBufferSize = 16; // Rx buffer size 16 (Non-Blocking mode)
config.hw_config = &hw_config;
