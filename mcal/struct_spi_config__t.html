Detailed Description

Spi configuration structure.

Example to configure SPI:

SpiHWConfig_t hw_config = {0};
hw_config.miso.port = PCF_PORT_0;
hw_config.miso.pin = PCF_PIN_4;
hw_config.mosi.port = PCF_PORT_0;
hw_config.mosi.pin = PCF_PIN_5;
hw_config.sck.port = PCF_PORT_0;
hw_config.sck.pin = PCF_PIN_11;
SpiConfig_t config = {0};
config.id = PCF_SPI4_ID;
config.mode = enSpiMode_Master; // Maaster mode
config.baudrate.unit = enFrequencyUnit_KHz; // baudrate uint to KHz
config.baudrate.value = 100; // baudrate value set to 100KHz
config.clockMode = enSpiClockMode_1; // Spi clock mode 1
config.frameSize = enSpiFrameSize_8; // Spi frame size 8
config.bitOrder = enSpiBitOrder_Msb; // Bit order MSB
config.maxSlaves = 4; // Max 4 slave can register
config.txBufferSize = 16; // Tx buffer size 16 (Non-Blocking mode)
config.rxBufferSize = 16; // Rx buffer size 16 (Non-Blocking mode)
// Must include <pcf/memory/memory.h> in order to utilize the @pcf/core memory APIs.
config.memoryConfig.memAlloc = pcf_Memory_allocate; // Using @pcf/core memory allocater.
config.memoryConfig.memFree = pcf_Memory_free; // Using @pcf/core memory free
config.hardwareCfg = &hw_config;
