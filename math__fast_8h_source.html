18 // Split a float into sign, exponent, and mantissa.
19 // sign = 0 for positive floats and 1 for negative floats
20 // exp (unbiased) = [0, 255]s
21 // mantissa = [0, (1<<23)-1]
22 PCF_INLINE static pcf_status_t pcf_data_Math_splitFloat(float fvalue, uint32_t * sign, uint32_t * exp, uint32_t * mantissa) {
23 PCF_ASSERT_RETURN(sign != PCF_NULL && exp != PCF_NULL && mantissa != PCF_NULL, PCF_STATUS_NULL_ARG);
24 uint32_t ivalue;
25 ivalue = PCF_FLOAT_2_INT(fvalue);
26 *sign = (ivalue >> 31) != 0 ? 1 : 0;
27 *exp = (ivalue >> 23) & 255;
28 *mantissa = ivalue & ((1 << 23) - 1);
29 return PCF_STATUS_OK;
30 }
