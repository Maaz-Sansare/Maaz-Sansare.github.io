#include <pcf/collections/buffer/buffer.h>
#include <pcf/memory/memory.h>
#include <stdio.h>
Buffer_t* txBuffer;
Buffer_t* rxBuffer;
uint8_t data[] = "Hello world!";
uint8_t dataStored[sizeof(data)];
uint8_t data2Read[sizeof(data)];
uint32_t FailCount = 0 , PassCount = 0;
pcf_size_t count ;
/* private function to check data in the buffer is correct*/
static void pcf_check_data(pcf_byte_t* compare , pcf_byte_t* data , pcf_size_t size){
uint8_t i;
for( i = 0 ; i < size ; i ++){
if(compare[i] != data [i]){
FailCount++;
break;
}
}
if( i == size ){
PassCount++;
}
}
int main(void)
{
pcf_status_t status;
pcf_size_t actual_size = 0 , size , count;
size = sizeof(data);
//Memory configurations
MemoryConfig_t memoryConfig;
memoryConfig.memAlloc = pcf_Memory_allocate;
memoryConfig.memFree = pcf_Memory_free;
memoryConfig.memCopy = pcf_Memory_copy;
memoryConfig.memMove = pcf_Memory_move;
BufferConfig_t txrb_config;
txrb_config.elementSize = enBufferElementSize_Byte; //Element size of data to be read or write
txrb_config.size = 64U; //Size of the buffer
txrb_config.memoryConfig = &memoryConfig;
//Creating a new buffer
txBuffer = pcf_Buffer_new(&txrb_config, &status);
if(!PCF_S_IS_SUCCESS(status)){
printf("Creating a new buffer failed..");
while(1);
}
BufferConfig_t rxrb_config;
rxrb_config.elementSize = enBufferElementSize_Byte; //Element size of data to be read or write
rxrb_config.size = 64U; //Size of the buffer
rxrb_config.memoryConfig = &memoryConfig;
//Creating another new buffer
rxBuffer = pcf_Buffer_new(&rxrb_config, &status);
if(!PCF_S_IS_SUCCESS(status)){
printf("Creating a new buffer failed..");
while(1);
}
actual_size = 0;
// copying from data -> txBuffer
pcf_Buffer_write(txBuffer, data, size , &actual_size);
if(actual_size != size){
printf("Writing into txBuffer failed..");
while(1);
}
// copying from txBuffer -> dataStored
for(uint8_t i = 0 ; i < size ; i++ ){
pcf_byte_t *byte;
pcf_Buffer_getReadAddress(txBuffer, (pcf_generic_t *)&byte);
dataStored[i] = (*byte);
pcf_Buffer_skip(txBuffer,1, PCF_NULL);
}
// copying from data -> rxBuffer
for(uint8_t i = 0 ; i < size ; i++ ){
pcf_byte_t *byte;
pcf_Buffer_getWriteAddress(rxBuffer, (pcf_generic_t *)&byte);
*byte = data[i];
pcf_Buffer_advance(rxBuffer, 1, PCF_NULL);
}
// copying from rxBuffer -> data2Read
pcf_Buffer_getFullCount(rxBuffer, &count);
if(count > 0){
pcf_Buffer_read(rxBuffer, data2Read, size, &actual_size);
}
// checking whether data is correct ot not
pcf_check_data(data , dataStored , size);
pcf_check_data(data , data2Read , size);
if(FailCount > 0){
printf("Copy unsuccessful\n");
}else if(PassCount > 0){
printf("Copied successfully\n");
}
}
