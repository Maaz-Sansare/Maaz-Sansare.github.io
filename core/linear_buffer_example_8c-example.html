#include <pcf/collections/linear_buffer/linear_buffer.h>
#include <pcf/memory/memory.h>
#include <stdio.h>
LinearBuffer_t* txBuffer;
LinearBuffer_t* rxBuffer;
uint8_t data[] = "Hello world!";
uint8_t dataStored[sizeof(data)];
uint8_t data2Read[sizeof(data)];
uint32_t FailCount = 0 , PassCount = 0;
pcf_size_t count ;
/* private function to check data in the buffer is correct*/
static void pcf_check_data(pcf_byte_t* compare , pcf_byte_t* data , pcf_size_t size){
uint8_t i;
for( i = 0 ; i < size ; i ++){
if(compare[i] != data [i]){
FailCount++;
break;
}
}
if( i == size ){
PassCount++;
}
}
int main(void)
{
pcf_status_t status;
pcf_size_t actual_size = 0 , size , count;
size = sizeof(data);
Data_t write_data, read_data;
write_data.data = data;
write_data.length = sizeof(data);
read_data.data = data2Read;
read_data.length = sizeof(data);
LinearBufferConfig_t txrb_config;
txrb_config.size = 64U; //Size of the buffer
txrb_config.memAlloc = pcf_Memory_alloc;
txrb_config.memFree = pcf_Memory_free;
//Creating a new buffer
txBuffer = pcf_LinearBuffer_new(&txrb_config, &status);
if(!PCF_S_IS_SUCCESS(status)){
printf("Creating a new buffer failed..");
while(1);
}
LinearBufferConfig_t rxrb_config;
rxrb_config.size = 64U; //Size of the buffer
rxrb_config.memAlloc = pcf_Memory_alloc;
rxrb_config.memFree = pcf_Memory_free;
//Creating another new buffer
rxBuffer = pcf_LinearBuffer_new(&rxrb_config, &status);
if(!PCF_S_IS_SUCCESS(status)){
printf("Creating a new buffer failed..");
while(1);
}
actual_size = 0;
// copying from data -> txBuffer
pcf_LinearBuffer_write(txBuffer, write_data, &actual_size);
if(actual_size != size){
printf("Writing into txBuffer failed..");
while(1);
}
// copying from txBuffer -> dataStored
for(uint8_t i = 0 ; i < size ; i++ ){
pcf_byte_t *byte;
pcf_LinearBuffer_getReadAddress(txBuffer, (pcf_generic_t *)&byte);
dataStored[i] = (*byte);
pcf_LinearBuffer_skip(txBuffer,1, PCF_NULL);
}
// copying from data -> rxBuffer
for(uint8_t i = 0 ; i < size ; i++ ){
pcf_byte_t *byte;
pcf_LinearBuffer_getWriteAddress(rxBuffer, (pcf_generic_t *)&byte);
*byte = data[i];
pcf_LinearBuffer_advance(rxBuffer, 1, PCF_NULL);
}
// copying from rxBuffer -> data2Read
pcf_LinearBuffer_getFullCount(rxBuffer, &count);
if(count > 0){
pcf_LinearBuffer_read(rxBuffer, read_data, &actual_size);
}
// checking whether data is correct ot not
pcf_check_data(data , dataStored , size);
pcf_check_data(data , data2Read , size);
if(FailCount > 0){
printf("Copy unsuccessful\n");
}else if(PassCount > 0){
printf("Copied successfully\n");
}
}
