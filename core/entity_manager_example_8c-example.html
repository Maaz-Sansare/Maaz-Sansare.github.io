entity_manager_example.c
#include <pcf/entity_manager/entity_manager.h>
#include <pcf/memory/memory.h>
int main(void){
pcf_status_t status;
pcf_size_t num_entities;
Entity_t* entity = PCF_NULL;
typedef struct{
uint16_t rpm;
uint16_t power;
} Motor_data;
Motor_data* data = pcf_Memory_alloc(sizeof(Motor_data), &status);
if(!PCF_S_IS_SUCCESS(status)){
return -1;
}
if(data == PCF_NULL){
return -1;
}
data->rpm = 100;
data->power = 500;
EntityManagerConfig_t config;
config.max_entities = 2; //Configuring Entity manager to have maximum capacity of 2
config.memAlloc = pcf_Memory_alloc;
config.memRealloc = pcf_Memory_realloc;
config.memFree = pcf_Memory_free;
//Initializing Entity manager with the configurations
status = pcf_EntityManager_init(&config);
if(!PCF_S_IS_SUCCESS(status)){
return -1;
}
Entity_t motor1;
motor1.id = 1;
motor1.type = (EntityType_t)((uint32_t)0x7A697344);
motor1.base = (pcf_generic_t)data;
motor1.messageHandler = PCF_NULL;
//Adding entity to the entity manager
status = pcf_EntityManager_addEntity(&motor1);
if(!PCF_S_IS_SUCCESS(status)){
return -1;
}
//Getting number of entity present in the entity manager
status = pcf_EntityManager_getEntityCount(&num_entities);
if(!PCF_S_IS_SUCCESS(status)){
return -1;
}
//Getting the entity from the manager of given type and at given index
entity = pcf_EntityManager_getEntity((EntityType_t)((uint32_t)0x7A697344), 1, &status);
if(entity == PCF_NULL){
return -1;
}
if(!PCF_S_IS_SUCCESS(status)){
return -1;
}
//Deinitialising the entity manager
status = pcf_EntityManager_deInit();
if(!PCF_S_IS_SUCCESS(status)){
return -1;
}
return 0;
}
