/*
* -------------------------------------------------------------
* Example : Assertion and debug using @pcf/core debug & assert
* -------------------------------------------------------------
*/
#include <pcf/assert/assert.h>
#include <pcf/debug/debug.h>
#include <stdio.h>
static uint8_t __assertFailCounter = 0;
// Creating debug_putchar of signature "DebugPutchar_fp"
// DebugPutchar_fp defined in <pcf/debug/types.h>
static pcf_status_t __debug_putchar(uint8_t ch){
// using putchar function to write message to the console
if(putchar(ch) == EOF){
return PCF_S_FATAL;
}
return PCF_S_SUCCESS;
}
static void __assertFaultHandler(void){
__assertFailCounter++;
}
int main(){
pcf_status_t status;
DebugConfig_t debugConfig;
debugConfig.putchar = __debug_putchar;
status = pcf_Debug_init(&debugConfig); // Initialize debug to write message
if(status != PCF_S_SUCCESS){ // checking status
while(1); // Initialization failed
}
AssertConfig_t assertConfig;
assertConfig.faultHandler = __assertFaultHandler;
status = pcf_Assert_init(&assertConfig); // Initialize assert
if(status != PCF_S_SUCCESS){ // checking status
while(1); // Initialization failed
}
pcf_size_t delta = 4;
pcf_size_t expected = 60;
pcf_size_t actual = 60;
pcf_size_t threshold = 80;
PCF_ASSERT_EQUAL(expected, actual); // will pass as expected is equal to actual value
PCF_ASSERT_EQUAL((expected + 1), actual); // will fail as (expected + 1) is not equal to actual value
PCF_ASSERT_LESS_THAN(threshold, actual); // will pass as actual value is less than threshold value
PCF_ASSERT_LESS_THAN(expected, actual); // will pass as actual value is equal to expected value
PCF_ASSERT_LESS_OR_EQUAL(threshold, actual); // will pass as actual value is less and not equal to threshold value
PCF_ASSERT_LESS_OR_EQUAL(expected, actual); // will fail as actual value is equal to expected value
expected = 57;
PCF_ASSERT_WITHIN(delta, expected, actual); // will pass as actual(60) is within the range(57+4 & 57-4)
expected = 7;
PCF_ASSERT_WITHIN(delta, expected, actual); // will fail as actual(60) is not within the range(7+4 & 7-4)
int64_t int_threshold = 50;
int64_t int_actual = 60;
PCF_ASSERT_GREATER_THAN_INT64(int_threshold, int_actual); // will pass as int_actual is greater than int_threshold value
PCF_ASSERT_LESS_THAN(int_threshold, int_actual); // will fail as int_actual is greater than int_threshold value
printf("\nAssertion fail counter = %d\n",__assertFailCounter); // printing __assertFailCounter.
return 0;
}
