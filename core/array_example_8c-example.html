#include <pcf/collections/array/array.h>
#include <pcf/memory/memory.h>
#include <stdio.h>
int mycompare(const void* a, const void* b)
{
int va = *(const int*)a;
int vb = *(const int*)b;
return (va > vb) - (va < vb);
}
CollectionPtrComparer_fp compare = mycompare;
int main(){
pcf_status_t status;
pcf_size_t size;
//Memory configuratons
MemoryConfig_t memoryConfig;
memoryConfig.memCalloc = pcf_Memory_calloc;
memoryConfig.memFree = pcf_Memory_free;
memoryConfig.memCopy = pcf_Memory_copy;
memoryConfig.memMove = pcf_Memory_move;
ArrayConfig_t config;
config.capacity = 10; //The current capacity of the array
config.maxElementCount = 17; //Maximum capacity of the array
config.growthFactor = 2; //Capacity will increase in multiple of 2
config.memoryConfig = &memoryConfig;
//Creating a new array with the configurations
Array_t* arr = pcf_Array_new(&config, &status);
if(!PCF_S_IS_SUCCESS(status)){
printf("Creating array object failed..");
while(1);
}
int a = 50;
int c = 4;
int d = 16;
//Adding element at the end of array
status = pcf_Array_insertAtEnd(arr, (void*)a);
if(!PCF_S_IS_SUCCESS(status)){
printf("Inserting element at end failed..");
while(1);
}
//Adding element at the end of array
status = pcf_Array_insertAtEnd(arr, (void*)c);
if(!PCF_S_IS_SUCCESS(status)){
printf("Inserting element at end failed..");
while(1);
}
//Adding element at the end of array
status = pcf_Array_insertAtEnd(arr, (void*)d);
if(!PCF_S_IS_SUCCESS(status)){
printf("Inserting element at end failed..");
while(1);
}
//Getting number of elements present in the array
status = pcf_Array_size(arr, &size);
if(!PCF_S_IS_SUCCESS(status)){
printf("Getting array size failed..");
while(1);
}
//Getting array and printing it
const void* const* buffer = pcf_Array_buffer(arr);
for (size_t i = 0; i < size; i++) {
printf("unsortedbuffer[%d] = %d\n", i, (int)buffer[i]);
}
printf("\n");
//Sorting the array using helper function compare
status = pcf_Array_sort(arr, compare);
if(!PCF_S_IS_SUCCESS(status)){
printf("Sorting array failed..");
while(1);
}
//Getting array
buffer = pcf_Array_buffer(arr);
//Printing the sorted array
for (size_t i = 0; i < size; i++) {
printf("sortedbuffer[%d] = %d\n", i, (int)buffer[i]);
}
return 0;
}
