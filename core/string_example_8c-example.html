/*
* -----------------------------------------------------
* Example : String manipulation using @pcf/core string
* -----------------------------------------------------
*/
#include <pcf/string/string.h>
#include <stdio.h>
int main(){
pcf_status_t status;
printf("String manipulation example using @pcf/core string.\n");
// Creating new String_t object with specified data.
String_t* str = pcf_String_new_fromCstr("HeLlO, WoRlD!...");
if(!str){
printf("Failed to create a new String_t object.\n");
return -1;
}
// Getting String_t data.
pcf_cstring_t data = pcf_String_toCstr(str);
printf("Original string : '%s'\n",data); // Printing original data.
// Converting original string data into upper case
status = pcf_String_toUpper(str);
if(!PCF_S_IS_SUCCESS(status)){
printf("Failed to do the string data into uppercase.\n");
return -1;
}
// Converting uppercase string data into lowercase
data = pcf_String_toCstr(str);
printf("Uppercase string : '%s'\n",data); // Printing uppercase data.
status = pcf_String_toLower(str);
if(!PCF_S_IS_SUCCESS(status)){
printf("Failed to do the string data into lowercase.\n");
return -1;
}
data = pcf_String_toCstr(str);
printf("Lowercase string : '%s'\n",data); // Printing lowercase data.
// Deleting previously created String_t object
status = pcf_String_delete(str);
if(!PCF_S_IS_SUCCESS(status)){
printf("Failed to delete previously created String_t object.\n");
return -1;
}
return 0;
}
