memory_example.c
/*
* -----------------------------------------------------
* Example : Memory manipulation using @pcf/core memory
* -----------------------------------------------------
*/
#include <pcf/memory/memory.h>
#include <pcf/assert/assert.h>
#include <stdio.h>
#define BUFFER_SIZE 40
int main(){
pcf_status_t status;
printf("Memory manipulation example using @pcf/core memory.\n");
// Allocating memory using @pcf/core memory of size 'BUFFER_SIZE'.
pcf_byte_t * buffer = (pcf_byte_t*)pcf_Memory_allocate(BUFFER_SIZE,&status);
if(buffer == PCF_NULL || !PCF_S_IS_SUCCESS(status)){
printf("Allocation for buffer failed of size %d.\n",BUFFER_SIZE);
return -1;
}
printf("Allocated buffer of size %d.\n",BUFFER_SIZE);
// Setting allocated memory to '0xAA'.
status = pcf_Memory_set(buffer,0xAA,BUFFER_SIZE);
if(!PCF_S_IS_SUCCESS(status)){
printf("Failed to set allocated buffer value to '0xAA'.");
}else{
printf("Allocated buffer value has been set to '0XAA'.\n");
}
// Creating local buffer of size 'BUFFER_SIZE' and setting it to '0xAA'.
pcf_byte_t localBuffer[BUFFER_SIZE];
for(int i = 0; i < BUFFER_SIZE ; i++){
localBuffer[i] = 0xAA;
}
printf("local buffer of size %d has been set to '0xAA'.\n",BUFFER_SIZE);
// Comparing values of allocated and local buffer
status = pcf_Memory_compare(buffer,localBuffer,BUFFER_SIZE);
if(!PCF_S_IS_YES(status)){
printf("local and allocated buffer value is not equal.\n");
}else{
printf("local and allocated buffer value is equal.\n");
}
// Freeing allocated memory
status = pcf_Memory_free(buffer);
if(!PCF_S_IS_SUCCESS(status)){
return -1;
}
printf("Allocated buffer memory has been freed.\n");
return 0;
}
