/*
* -----------------------------------------------------
* Example : Memory manipulation using @pcf/core memory
* -----------------------------------------------------
*/
#include <pcf/memory/memory.h>
#include <pcf/assert/assert.h>
#include <stdio.h>
int main(){
pcf_status_t status;
pcf_size_t buf_size = 40; // Buffer size set to 40
printf("Memory manipulation example using @pcf/core memory.\n");
// Allocating memory using @pcf/core memory of size 'buf_size'.
pcf_byte_t * buffer = (pcf_byte_t*)pcf_Memory_allocate(buf_size,&status);
#if(PCF_DEBUG == PCF_YES)
PCF_ASSERT_NOT_NULL(buffer); // Halt if assertion failed
PCF_ASSERT_TRUE(PCF_S_IS_SUCCESS(status)); // Halt if assertion failed
#else
if(buffer == PCF_NULL || !PCF_S_IS_SUCCESS(status)){
printf("Allocation for buffer failed of size %d.\n",buf_size);
while(1);
}
#endif
printf("Allocated buffer of size %d.\n",buf_size);
// Setting allocated memory to '0xAA'.
status = pcf_Memory_set(buffer,0xAA,buf_size);
#if(PCF_DEBUG == PCF_YES)
PCF_ASSERT_TRUE(PCF_S_IS_SUCCESS(status)); // Halt if assertion failed
#else
if(!PCF_S_IS_SUCCESS(status)){
printf("Failed to set allocated buffer value to '0xAA'.");
while(1);
}
#endif
printf("Allocated buffer value has been set to '0XAA'.\n");
// Creating local buffer of size 'buf_size' and setting it to '0xAA'.
pcf_byte_t localBuffer[buf_size];
for(int i = 0; i < buf_size ; i++){
localBuffer[i] = 0xAA;
}
printf("local buffer of size %d has been set to '0xAA'.\n",buf_size);
// Comparing values of allocated and local buffer
status = pcf_Memory_compare(buffer,localBuffer,buf_size);
#if(PCF_DEBUG == PCF_YES)
PCF_ASSERT_TRUE(PCF_S_IS_YES(status)); // Halt if assertion failed
#else
if(!PCF_S_IS_YES(status)){
printf("local and allocated buffer value is not equal.\n");
while(1);
}
#endif
printf("local and allocated buffer value is equal.\n");
// Freeing allocated memory
status = pcf_Memory_free(buffer);
#if(PCF_DEBUG == PCF_YES)
PCF_ASSERT_TRUE(PCF_S_IS_SUCCESS(status)); // Halt if assertion failed
#else
if(!PCF_S_IS_SUCCESS(status)){
while(1);
}
#endif
printf("Allocated buffer memory has been freed.\n");
return 0;
}
