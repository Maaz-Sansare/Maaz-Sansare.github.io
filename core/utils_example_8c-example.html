/*
* -------------------------------------------
* Example : Utils apis using @pcf/core utils
* -------------------------------------------
*/
#include <pcf/utils/utils.h>
#include <stdio.h>
int main(){
int64_t num = 689724;
char str[15];
// Converting 'num' signed integer value into string.
if(pcf_Utils_iToA(num,str,8) > 0){
printf("iToA output : '%s'\n",str); // Printing converted 'num' integer value.
}else{
printf("Signed integer value to string coversion failed.\n"); // Printing error message.
}
float fvalue = 1.23456e+10f;
// Converting 'fvalue' float value into string.
if(pcf_Utils_fToA_exp(fvalue, str, 6, enPCFBoolean_True, enPCFBoolean_True) > 0){
printf("fToA output : '%s'\n",str);
}else{
printf("Float value to string coversion failed.\n"); // Printing error message.
}
double dvalue = -1.23456e-10;
// Converting 'dvalue' double value into string.
if(pcf_Utils_dToA_exp(dvalue, str, 0, enPCFBoolean_False, enPCFBoolean_True) > 0){
printf("dToA output : '%s'\n",str);
}else{
printf("Double value to string coversion failed.\n"); // Printing error message.
}
unsigned uvalue = 0xAB;
// Converting 'uvalue' unsigned integer value into string.
if(pcf_Utils_uToA(uvalue,str, sizeof(str), 16, enPCFBoolean_True) > 0){
printf("uToA output : '%s'\n",str);
}else{
printf("Unsigned integer value to string coversion failed.\n"); // Printing error message.
}
return 0;
}
